💾 Database API (db.js)

IndexedDB wrapper for all data operations.

INIT DB.ready()

Initialize database connection. Auto-creates stores.

CREATE DB.add(storeName, data)

Add new record. Auto-generates ID if not provided.

READ DB.get(storeName, id)

Get record by ID.

READ ALL DB.getAll(storeName)

Get all records from a store.

UPDATE DB.update(storeName, data)

Update existing record.

DELETE DB.delete(storeName, id)

Remove record by ID.

// Example Usage
await DB.ready();
await DB.add('students', { name: 'John', class: '10A' });
const students = await DB.getAll('students');

🔐 Authentication API (auth.js)

SHA-256 password hashing with session management.

// Login
const result = await Auth.login(username, password, role);
if (result.success) {
  Auth.redirectToDashboard(role);
}

// Check auth
if (!Auth.isLoggedIn()) { /* redirect */ }
Auth.requireRole(['admin', 'developer']);

// Get current user
const user = Auth.currentUser();
// { userId, username, name, role, linkedId }

// Logout
Auth.logout();

📊 Database Stores

users

{ id, username, passwordHash, role, name, linkedId }

students

{ id, name, rollNo, class, dob, guardianName, phone }

faculty

{ id, name, email, phone, subjects[], assignedClasses[] }

attendance

{ id, studentId, date, status, classId, markedBy }

marks

{ id, studentId, subjectId, examType, marks, maxMarks }

fees

{ id, studentId, totalAmount, paidAmount, balance, status }
